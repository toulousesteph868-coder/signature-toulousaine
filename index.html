    <script>

        const jours = ["Lundi", "Mardi", "Mercredi", "Jeudi", "Vendredi", "Samedi", "Dimanche"];

        const headerRow = document.getElementById('headerRow');

        const tableBody = document.getElementById('tableBody');

        const bookingForm = document.getElementById('booking-form');

        const summaryText = document.getElementById('selection-summary');

 

        jours.forEach(j => {

            let th = document.createElement('th');

            th.innerText = j;

            headerRow.appendChild(th);

        });

 

        for (let h = 8; h <= 25.5; h += 0.5) {

            let hInt = Math.floor(h);

            let m = (h % 1 === 0) ? "00" : "30";

            let displayH = hInt >= 24 ? hInt - 24 : hInt;

            let timeStr = `${displayH.toString().padStart(2, '0')}h${m}`;

           

            let tr = document.createElement('tr');

            let timeTd = document.createElement('td');

            timeTd.className = "time-col";

            timeTd.innerText = timeStr;

            tr.appendChild(timeTd);

 

            jours.forEach((jour, idx) => {

                let td = document.createElement('td');

                let isAvail = false;

               

                // Logique basée sur l'image

                if (idx < 5) { // Semaine

                    if (h >= 18.5 && h < 24) isAvail = true;

                } else { // Weekend

                    if ((h >= 8.5 && h < 15) || (h >= 23 && h <= 25.5)) isAvail = true;

                }

 

                if (isAvail) {

                    td.className = "available";

                    td.innerText = "Libre";

                    td.onclick = function() {

                        document.querySelectorAll('.selected').forEach(el => el.classList.remove('selected'));

                        this.classList.add('selected');

                        showForm(jour, timeStr);

                    };

                } else {

                    td.className = "closed";

                    td.innerText = "-";

                }

                tr.appendChild(td);

            });

            tableBody.appendChild(tr);

        }

 

        function showForm(jour, heure) {

            bookingForm.style.display = 'block';

            summaryText.innerText = `Créneau : ${jour} à ${heure}`;

            bookingForm.scrollIntoView({ behavior: 'smooth' });

        }

 

        function submitBooking() {

            const name = document.getElementById('userName').value;

            if(!name) return alert("Merci de préciser votre nom.");

            alert(`Merci ${name} ! Votre demande pour le ${summaryText.innerText} a été simulée. Dans une version réelle, vous recevriez un mail ou une notification.`);

            bookingForm.style.display = 'none';

        }

    </script>
